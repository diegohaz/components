import {Field} from '@salad-ui/field';
import {TextInput} from '@salad-ui/text-input';
export const scope = {Field, TextInput};

Fields provide users with assistance and feedback as they enter data.

## Installation

```bash
npm i @salad-ui/field
```

## Usage

```jsx
import {Field} from '@salad-ui/field';
import {TextInput} from '@salad-ui/text-input';

<Field
  label="Email"
  help={
    <>
      We'll contact you from time-to-time with progress on <code>salad-ui</code>
      .
    </>
  }
>
  <TextInput placeholder="john.smith@example.com" isFullWidth />
</Field>;
```
