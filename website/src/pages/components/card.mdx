A Card groups related information.

## Installation

```bash
npm i @salad-ui/card
```

## Usage

```jsx
import {Card} from '@salad-ui/card';
import {Title, Body} from '@salad-ui/typography';

<Card>
  <Title size="small">Change your site address</Title>
  <Body>Register a new domain or change your site's address.</Body>
</Card>
```
