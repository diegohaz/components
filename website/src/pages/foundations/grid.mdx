import styled from 'styled-components';
import {grid, unit, Grid, Unit} from '@salad-ui/grid';
export const scope = {styled, grid, unit, Grid, Unit};

## Installation

```bash
npm install @salad-ui/grid
```

## Usage

### Mixins

```jsx inline=false
import styled from 'styled-components';
import {grid, unit} from '@salad-ui/grid';

const FeaturePanel = styled.div`
  ${grid({reverse: {mobile: true, tablet: false}})}
`;

const Feature = styled.div`
  ${unit({size: {mobile: 1, tablet: 1 / 3, desktop: 1 / 5}})}
`;

render(
  <FeaturePanel>
    <Feature>A</Feature>
    <Feature>B</Feature>
    <Feature>C</Feature>
    <Feature>D</Feature>
    <Feature>E</Feature>
  </FeaturePanel>,
);
```

### Components

```jsx
import {Grid, Unit} from '@salad-ui/grid';

<Grid reverse={true}>
  <Unit size={{mobile: 1, tablet: 1 / 3, desktop: 1 / 5}}>A</Unit>
  <Unit size={{mobile: 1, tablet: 1 / 3, desktop: 1 / 5}}>B</Unit>
  <Unit size={{mobile: 1, tablet: 1 / 3, desktop: 1 / 5}}>C</Unit>
  <Unit size={{mobile: 1, tablet: 1 / 3, desktop: 1 / 5}}>D</Unit>
  <Unit size={{mobile: 1, tablet: 1 / 3, desktop: 1 / 5}}>E</Unit>
</Grid>;
```
