## Installation

```bash
npm install @salad-ui/typography
```

## Usage

```jsx
import {Subtitle} from '@salad-ui/typography';

<>
  <Subtitle>The quick brown fox jumps over the lazy dog</Subtitle>
  <Subtitle isSmall>The quick brown fox jumps over the lazy dog</Subtitle>
</>;
```
